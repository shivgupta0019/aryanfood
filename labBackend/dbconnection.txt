-- DROP TABLE main_labtest;
CREATE TABLE main_labtest (
    id INT AUTO_INCREMENT PRIMARY KEY,
    labtest_code VARCHAR(20) UNIQUE,
    labtest_name VARCHAR(150) NOT NULL,
    location VARCHAR(150) NOT NULL,
    labtest_type JSON NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 1st ETO Testing

CREATE TABLE eto (
    id INT AUTO_INCREMENT PRIMARY KEY,
    eto_concentration VARCHAR(50) NOT NULL,   -- e.g. 600 mg/L
    exposure_time VARCHAR(50) NOT NULL,       -- e.g. 4 hours
    temperature VARCHAR(50) NOT NULL,         -- e.g. 54°C
    humidity VARCHAR(50) NOT NULL,            -- e.g. 60%
    residue_level VARCHAR(50) NOT NULL,       -- e.g. < 10 ppm
    labtest_code VARCHAR(20) NOT NULL
);

-- 2nd micro testing

CREATE TABLE micro (
    id INT AUTO_INCREMENT PRIMARY KEY,
    total_plate_count VARCHAR(50) NOT NULL,     -- e.g. < 100 CFU/g
    yeast_mold VARCHAR(50) NOT NULL,            -- e.g. < 10 CFU/g
    e_coli VARCHAR(50) NOT NULL,                -- e.g. Absent/25g
    salmonella VARCHAR(50) NOT NULL,            -- e.g. Absent/25g
    coliform_bacteria VARCHAR(50) NOT NULL,     -- e.g. < 3 MPN/g
    labtest_code VARCHAR(20) NOT NULL
);

-- 3rd physical testing

CREATE TABLE physical (
    id INT AUTO_INCREMENT PRIMARY KEY,
    texture_analysis VARCHAR(100) NOT NULL,        -- e.g. Firm, 8.5 N
    color_measurement VARCHAR(100) NOT NULL,       -- e.g. L*=92
    moisture_content VARCHAR(50) NOT NULL,         -- e.g. 12.5%
    particle_size_shape VARCHAR(100) NOT NULL,     -- e.g. 50–100 µm
    viscosity VARCHAR(100) NOT NULL,               -- e.g. 450 mPa·s
    density_specific VARCHAR(100) NOT NULL,        -- e.g. 1.05 g/cm³
    water_activity VARCHAR(50) NOT NULL,           -- e.g. 0.87 aw
    labtest_code VARCHAR(20) NOT NULL
);

-- 4th chemical
CREATE TABLE chemical (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ph_level VARCHAR(20) NOT NULL,           -- e.g. 6.8
    ash_content VARCHAR(50) NOT NULL,        -- e.g. 1.2%
    protein_content VARCHAR(50) NOT NULL,    -- e.g. 18.5%
    fat_content VARCHAR(50) NOT NULL,        -- e.g. 5.3%
    peroxide_value VARCHAR(50) NOT NULL,     -- e.g. 2.1 meq/kg
    heavy_metals VARCHAR(100) NOT NULL,      -- e.g. Pb < 0.5 ppm
    labtest_code VARCHAR(20) NOT NULL
);

-- instrument table


CREATE TABLE instrument (
    id INT AUTO_INCREMENT PRIMARY KEY,

    instrument_name VARCHAR(150) NOT NULL,       -- e.g. HPLC Analyzer
    model VARCHAR(150) NOT NULL,                 -- e.g. Agilent 1260
    calibration_date DATE NOT NULL,             -- dd/mm/yyyy
    next_calibration_date DATE NOT NULL,        -- dd/mm/yyyy
    labtest_code VARCHAR(20) NOT NULL
);
